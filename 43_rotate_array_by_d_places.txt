class Solution:
    # T: O(n), S: O(1) 
    def reverse(self, nums, l, h):
        while l < h:
            nums[l], nums[h] = nums[h], nums[l]
            l += 1
            h -= 1

    def rotate(self, nums: List[int], k: int) -> None:
        k = k % len(nums)
        self.reverse(nums, 0, len(nums) - k - 1)
        self.reverse(nums, len(nums) - k, len(nums) - 1)
        self.reverse(nums, 0, len(nums) - 1)
